import { useState, useRef } from "react";

const AL = {listo:'🎯 Listo',inicio:'▶️ Inicio',termine:'🏁 Terminé',siguiente:'✅ Siguiente',intento:'📝 Intento',fallo:'❌ Fallo',reiniciar:'🔄 Reiniciar',saltar:'⏭️ Saltar',pausa:'⏸️ Pausa',ninguna:'— Ninguna'};
const fmt = s => { const m=Math.floor(s/60),sec=s%60; return m>0?m+"'"+String(sec).padStart(2,'0')+'"':sec+'"'; };
const STEPS_LABELS = ['LEYENDO EL ENTRENAMIENTO','IDENTIFICANDO EJERCICIOS','CALCULANDO TIMERS','GENERANDO PASOS Y TAPS','PREPARANDO SESIÓN'];

export default function ImportarIA() {
  const [screen, setScreen] = useState('input'); // input | loading | result
  const [text, setText] = useState('');
  const [loadingStep, setLoadingStep] = useState(-1);
  const [session, setSession] = useState(null);
  const [error, setError] = useState('');
  const [openEx, setOpenEx] = useState(0);
  const [sessName, setSessName] = useState('');
  const [saved, setSaved] = useState(false);
  const loadTimers = useRef([]);

  const animateLoading = () => {
    loadTimers.current.forEach(clearTimeout);
    loadTimers.current = [];
    setLoadingStep(0);
    STEPS_LABELS.forEach((_, i) => {
      const t = setTimeout(() => setLoadingStep(i), i * 800);
      loadTimers.current.push(t);
    });
  };

  const generate = async () => {
    if (!text.trim() || text.length < 20) return;
    setScreen('loading');
    setError('');
    setSession(null);
    setSaved(false);
    animateLoading();

    const prompt = `Eres un asistente experto en entrenamiento de escalada indoor. Convierte este entrenamiento en JSON ejecutable.

REGLAS:
- Cada bloque lógico = un ejercicio con pasos
- Tipos de pasos: "climby" (texto motivacional del entrenador), "timer" (segundos), "tap" (botón físico)
- tap: "uno" = 1 toque, "dos" = 2 toques. Valores: listo, inicio, termine, siguiente, intento, fallo, reiniciar, saltar, pausa, ninguna
- Descansos → timers
- Series X2/X3 → duplicar/triplicar ejercicios como "Serie 1", "Serie 2"
- Texto climby: motivacional, claro, como entrenador real hablando
- Porcentajes del máximo → texto explicativo en climby
- Cada ejercicio empieza con climby explicando qué es
- 1 toque = "termine" o "siguiente". 2 toques = "reiniciar" o "fallo"
- Si hay series dentro de un ejercicio, cada serie es un ejercicio separado

DEVUELVE SOLO JSON sin texto ni backticks:
{"nombre":"Nombre sesión","ejercicios":[{"nombre":"Nombre","pasos":[{"tipo":"climby","texto":"..."},{"tipo":"timer","segundos":45},{"tipo":"tap","uno":"termine","dos":"reiniciar"}]}]}

ENTRENAMIENTO:
${text}`;

    try {
      const response = await fetch("https://api.anthropic.com/v1/messages", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          model: "claude-sonnet-4-20250514",
          max_tokens: 4000,
          messages: [{ role: "user", content: prompt }]
        })
      });

      if (!response.ok) {
        const err = await response.json();
        throw new Error(err.error?.message || 'Error de API: ' + response.status);
      }

      const data = await response.json();
      const raw = data.content.map(c => c.text || '').join('');
      const clean = raw.replace(/```json\n?/g,'').replace(/```\n?/g,'').trim();
      const parsed = JSON.parse(clean);
      setSession(parsed);
      setSessName(parsed.nombre || 'Sesión importada');
      setOpenEx(0);
      setTimeout(() => setScreen('result'), 400);
    } catch(e) {
      setError(e.message || 'Error desconocido');
      setTimeout(() => setScreen('result'), 400);
    }
  };

  const saveSession = () => {
    if (!session) return;
    try {
      const EX_KEY = 'c7a_ex_v2';
      const SESS_KEY = 'c7a_sessions_v1';
      const saved_exs = JSON.parse(localStorage.getItem(EX_KEY)||'[]');
      const sessions = JSON.parse(localStorage.getItem(SESS_KEY)||'[]');
      const newIds = [];
      (session.ejercicios||[]).forEach(ex => {
        const newEx = {
          id: 'ia_'+Date.now()+'_'+Math.random().toString(36).substr(2,4),
          name: ex.nombre,
          desc: 'Importado con IA',
          steps: (ex.pasos||[]).map(p => {
            if(p.tipo==='climby') return {type:'climby',text:p.texto};
            if(p.tipo==='timer') return {type:'timer',seconds:p.segundos};
            if(p.tipo==='tap') return {type:'tap',one:p.uno||'siguiente',two:p.dos||'fallo'};
            return null;
          }).filter(Boolean)
        };
        saved_exs.unshift(newEx);
        newIds.push(newEx.id);
      });
      localStorage.setItem(EX_KEY, JSON.stringify(saved_exs));
      sessions.unshift({ id:'ia_sess_'+Date.now(), name:sessName, exerciseIds:newIds });
      localStorage.setItem(SESS_KEY, JSON.stringify(sessions));
      setSaved(true);
    } catch(e) { setError('Error al guardar: '+e.message); }
  };

  const totalPasos = (session?.ejercicios||[]).reduce((a,e)=>a+(e.pasos?.length||0),0);
  const totalTimers = (session?.ejercicios||[]).reduce((a,e)=>a+(e.pasos?.filter(p=>p.tipo==='timer').length||0),0);

  // ── INPUT ──
  if (screen === 'input') return (
    <div style={{background:'#0f0f0f',minHeight:'100vh',color:'#f2ece4',fontFamily:'system-ui,sans-serif'}}>
      <div style={{padding:'20px 20px 14px',borderBottom:'1px solid rgba(255,255,255,0.08)',display:'flex',alignItems:'center',gap:12}}>
        <div style={{flex:1}}>
          <div style={{fontFamily:'monospace',fontSize:20,fontWeight:900,letterSpacing:4,color:'#f2ece4'}}>IMPORTAR CON IA</div>
          <div style={{fontSize:10,letterSpacing:2,color:'#666',marginTop:2}}>GENERADOR DE SESIONES</div>
        </div>
        <div style={{display:'flex',alignItems:'center',gap:6,background:'rgba(201,169,110,0.1)',border:'1px solid rgba(201,169,110,0.25)',borderRadius:20,padding:'5px 12px'}}>
          <div style={{width:6,height:6,borderRadius:'50%',background:'#c9a96e'}}/>
          <span style={{fontSize:11,letterSpacing:2,color:'#c9a96e'}}>CLAUDE</span>
        </div>
      </div>

      <div style={{padding:'20px 20px 140px'}}>
        <div style={{fontSize:10,letterSpacing:2,color:'#666',marginBottom:8}}>PEGA O ESCRIBE TU ENTRENAMIENTO</div>
        <textarea
          value={text}
          onChange={e=>setText(e.target.value)}
          style={{width:'100%',background:'#1a1a1a',border:'1px solid rgba(255,255,255,0.08)',borderRadius:14,padding:16,color:'#f2ece4',fontFamily:'system-ui,sans-serif',fontSize:14,lineHeight:1.7,outline:'none',resize:'none',minHeight:320}}
          placeholder={`Pega aquí tu entrenamiento tal como lo tienes escrito. Por ejemplo:

DÍA 1 - SEMANA 1

CALENTAMIENTO:
Movilidad general y estiramientos
Antagonistas con goma 45'' trabajo 10'' descanso X2
- Manguito rotador
- Pull over diagonal

Bíceps con goma:
15 rep 10'' relax
30 rep 20'' relax
50 rep 30'' relax
X2

Dominadas test 2' descanso...`}
        />
        <div style={{textAlign:'right',fontSize:11,color:'#555',marginTop:6}}>{text.length} caracteres</div>

        <div style={{background:'#1a1a1a',border:'1px solid rgba(255,255,255,0.07)',borderRadius:12,padding:14,marginTop:16}}>
          <div style={{fontSize:10,letterSpacing:2,color:'#555',marginBottom:10}}>CONSEJOS</div>
          {[
            ['✓','Escribe con lenguaje natural, como en papel. Sin formato especial.'],
            ['✓','Incluye tiempos, repeticiones, series y descansos.'],
            ['✓','Las series X2 o X3 se duplican automáticamente.'],
            ['✓','Puedes pegar de WhatsApp, notas o escribir aquí directamente.'],
          ].map(([icon,tip],i)=>(
            <div key={i} style={{display:'flex',gap:8,marginBottom:7}}>
              <span style={{color:'#c9a96e',fontSize:14}}>{icon}</span>
              <span style={{fontSize:12,color:'#666',lineHeight:1.5}}>{tip}</span>
            </div>
          ))}
        </div>
      </div>

      <div style={{position:'fixed',bottom:0,left:0,right:0,padding:'16px 20px 30px',background:'linear-gradient(transparent,#0f0f0f 30%)'}}>
        <button
          onClick={generate}
          disabled={text.length < 20}
          style={{width:'100%',background:text.length<20?'#333':'#c9a96e',color:text.length<20?'#555':'#0f0f0f',border:'none',borderRadius:14,padding:18,fontFamily:'monospace',fontSize:16,fontWeight:900,letterSpacing:4,cursor:text.length<20?'not-allowed':'pointer',transition:'all .2s'}}
        >
          ✦ GENERAR SESIÓN
        </button>
      </div>
    </div>
  );

  // ── LOADING ──
  if (screen === 'loading') return (
    <div style={{background:'#0f0f0f',minHeight:'100vh',color:'#f2ece4',fontFamily:'system-ui,sans-serif',display:'flex',flexDirection:'column',alignItems:'center',justifyContent:'center',padding:'40px 20px',textAlign:'center'}}>
      <div style={{fontSize:56,marginBottom:24,animation:'none'}}>🧗</div>
      <div style={{fontFamily:'monospace',fontSize:22,fontWeight:900,letterSpacing:3,color:'#c9a96e',marginBottom:8}}>CLIMBY ANALIZANDO</div>
      <div style={{fontSize:13,color:'#666',lineHeight:1.6,maxWidth:280,marginBottom:28}}>Convirtiendo tu entrenamiento en pasos ejecutables...</div>
      <div style={{display:'flex',flexDirection:'column',gap:8,width:'100%',maxWidth:320}}>
        {STEPS_LABELS.map((lbl,i)=>{
          const done = i < loadingStep;
          const active = i === loadingStep;
          return (
            <div key={i} style={{display:'flex',alignItems:'center',gap:10,padding:'10px 14px',background:done?'rgba(93,184,122,0.06)':active?'rgba(201,169,110,0.06)':'#1a1a1a',border:`1px solid ${done?'rgba(93,184,122,0.4)':active?'rgba(201,169,110,0.4)':'rgba(255,255,255,0.07)'}`,borderRadius:10,transition:'all .4s'}}>
              <div style={{width:8,height:8,borderRadius:'50%',background:done?'#5db87a':active?'#c9a96e':'rgba(255,255,255,0.1)',flexShrink:0,transition:'background .3s'}}/>
              <span style={{fontSize:12,color:done?'#5db87a':active?'#c9a96e':'#555',letterSpacing:1}}>{lbl}</span>
            </div>
          );
        })}
      </div>
    </div>
  );

  // ── RESULT ──
  return (
    <div style={{background:'#0f0f0f',minHeight:'100vh',color:'#f2ece4',fontFamily:'system-ui,sans-serif'}}>
      <div style={{padding:'20px 20px 14px',borderBottom:'1px solid rgba(255,255,255,0.08)',display:'flex',alignItems:'center',gap:12}}>
        <button onClick={()=>setScreen('input')} style={{color:'#c9a96e',fontSize:22,background:'none',border:'none',cursor:'pointer'}}>←</button>
        <div style={{fontFamily:'monospace',fontSize:20,fontWeight:900,letterSpacing:3}}>RESULTADO</div>
      </div>

      <div style={{padding:'20px 20px 140px'}}>
        {error ? (
          <div style={{background:'rgba(224,92,92,0.08)',border:'1px solid rgba(224,92,92,0.25)',borderRadius:12,padding:16,marginBottom:16}}>
            <div style={{fontFamily:'monospace',fontSize:14,fontWeight:700,color:'#e05c5c',marginBottom:6}}>ERROR AL GENERAR</div>
            <div style={{fontSize:13,color:'#666',lineHeight:1.5}}>{error}</div>
            <button onClick={()=>setScreen('input')} style={{marginTop:12,background:'#1a1a1a',border:'1px solid rgba(255,255,255,0.1)',borderRadius:8,padding:'10px 16px',color:'#c9a96e',fontFamily:'monospace',fontSize:12,letterSpacing:1,cursor:'pointer'}}>← VOLVER</button>
          </div>
        ) : session && (
          <>
            <div style={{background:'#1a1a1a',border:'1px solid rgba(201,169,110,0.2)',borderRadius:16,padding:20,marginBottom:20}}>
              <div style={{display:'inline-flex',alignItems:'center',gap:6,background:'rgba(93,184,122,0.1)',border:'1px solid rgba(93,184,122,0.3)',borderRadius:20,padding:'4px 10px',marginBottom:12}}>
                <span style={{fontSize:10,letterSpacing:2,color:'#5db87a'}}>✓ GENERADO POR IA</span>
              </div>
              <input
                value={sessName}
                onChange={e=>setSessName(e.target.value)}
                style={{display:'block',width:'100%',background:'#242424',border:'1px solid rgba(255,255,255,0.08)',borderRadius:10,padding:'12px 14px',color:'#f2ece4',fontFamily:'monospace',fontSize:18,fontWeight:700,outline:'none',marginBottom:12}}
              />
              <div style={{display:'flex',gap:20}}>
                {[[(session.ejercicios||[]).length,'EJERCICIOS'],[totalPasos,'PASOS'],[totalTimers,'TIMERS']].map(([n,l],i)=>(
                  <div key={i}>
                    <div style={{fontFamily:'monospace',fontSize:24,fontWeight:900,color:'#c9a96e'}}>{n}</div>
                    <div style={{fontSize:10,letterSpacing:1,color:'#555'}}>{l}</div>
                  </div>
                ))}
              </div>
            </div>

            <div style={{fontSize:10,letterSpacing:3,color:'#555',marginBottom:12}}>EJERCICIOS — toca para ver los pasos</div>

            {(session.ejercicios||[]).map((ex,i)=>(
              <div key={i} style={{background:'#1a1a1a',border:'1px solid rgba(255,255,255,0.07)',borderRadius:12,marginBottom:10,overflow:'hidden'}}>
                <div onClick={()=>setOpenEx(openEx===i?-1:i)} style={{padding:'14px 16px',display:'flex',alignItems:'center',gap:10,cursor:'pointer'}}>
                  <div style={{fontFamily:'monospace',fontSize:14,fontWeight:900,color:'#c9a96e',width:20}}>{i+1}</div>
                  <div style={{fontFamily:'monospace',fontSize:16,fontWeight:700,flex:1,letterSpacing:.5}}>{ex.nombre}</div>
                  <span style={{color:'#555',fontSize:14,transform:openEx===i?'rotate(180deg)':'none',transition:'transform .2s'}}>▾</span>
                </div>
                {openEx===i && (
                  <div style={{borderTop:'1px solid rgba(255,255,255,0.07)'}}>
                    {(ex.pasos||[]).map((p,j)=>{
                      const iconStyle={width:28,height:28,borderRadius:'50%',display:'flex',alignItems:'center',justifyContent:'center',fontSize:13,flexShrink:0};
                      if(p.tipo==='climby') return (
                        <div key={j} style={{display:'flex',alignItems:'flex-start',gap:10,padding:'10px 16px',borderBottom:'1px solid rgba(255,255,255,0.05)'}}>
                          <div style={{...iconStyle,background:'rgba(201,169,110,0.15)'}}>🗣️</div>
                          <div style={{fontSize:13,color:'#999',lineHeight:1.5,flex:1}}>{p.texto}</div>
                        </div>
                      );
                      if(p.tipo==='timer') return (
                        <div key={j} style={{display:'flex',alignItems:'center',gap:10,padding:'10px 16px',borderBottom:'1px solid rgba(255,255,255,0.05)'}}>
                          <div style={{...iconStyle,background:'rgba(93,184,122,0.15)'}}>⏱️</div>
                          <div style={{fontSize:13,color:'#999',flex:1}}>Timer</div>
                          <div style={{fontFamily:'monospace',fontSize:16,fontWeight:700,color:'#5db87a'}}>{fmt(p.segundos)}</div>
                        </div>
                      );
                      if(p.tipo==='tap') return (
                        <div key={j} style={{display:'flex',alignItems:'center',gap:10,padding:'10px 16px',borderBottom:'1px solid rgba(255,255,255,0.05)'}}>
                          <div style={{...iconStyle,background:'rgba(91,155,213,0.15)'}}>👆</div>
                          <div style={{fontSize:12,color:'#999',flex:1,lineHeight:1.5}}>
                            <span style={{color:'#5b9bd5'}}>1 toque: {AL[p.uno]||p.uno}</span><br/>
                            <span style={{color:'#e05c5c'}}>2 toques: {AL[p.dos]||p.dos}</span>
                          </div>
                        </div>
                      );
                      return null;
                    })}
                  </div>
                )}
              </div>
            ))}
          </>
        )}
      </div>

      {session && !error && (
        <div style={{position:'fixed',bottom:0,left:0,right:0,padding:'16px 20px 30px',background:'linear-gradient(transparent,#0f0f0f 30%)',display:'flex',flexDirection:'column',gap:8}}>
          {saved ? (
            <div style={{width:'100%',background:'rgba(93,184,122,0.12)',border:'1px solid rgba(93,184,122,0.3)',borderRadius:12,padding:16,textAlign:'center',fontFamily:'monospace',fontSize:14,letterSpacing:2,color:'#5db87a'}}>
              ✓ GUARDADO EN BIBLIOTECA
            </div>
          ) : (
            <button onClick={saveSession} style={{width:'100%',background:'#c9a96e',color:'#0f0f0f',border:'none',borderRadius:12,padding:16,fontFamily:'monospace',fontSize:16,fontWeight:900,letterSpacing:3,cursor:'pointer'}}>
              GUARDAR EN BIBLIOTECA
            </button>
          )}
          <button onClick={()=>setScreen('input')} style={{width:'100%',background:'#1a1a1a',border:'1px solid rgba(255,255,255,0.08)',borderRadius:12,padding:14,color:'#666',fontFamily:'monospace',fontSize:13,letterSpacing:2,cursor:'pointer'}}>
            ← VOLVER A EDITAR
          </button>
        </div>
      )}
    </div>
  );
}
